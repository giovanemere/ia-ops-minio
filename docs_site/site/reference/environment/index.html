    Script de Validaci贸n

    #!/bin/bash
# validate-env.sh

required_vars=(
    "MINIO_ROOT_USER"
    "MINIO_ROOT_PASSWORD"
    "DASHBOARD_PORT"
    "MINIO_CONSOLE_PORT"
    "MINIO_API_PORT"
)

missing_vars=()

for var in "${required_vars[@]}"; do
    if [ -z "${!var}" ]; then
        missing_vars+=("$var")
    fi
done

if [ ${#missing_vars[@]} -gt 0 ]; then
    echo "ERROR: Missing required environment variables:"
    printf '%s\n' "${missing_vars[@]}"
    exit 1
fi

echo " All required environment variables are set"
